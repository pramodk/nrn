147 rl_add_funmap_entry (char* name, Function* function)
148 {
149  if (funmap_entry + 2 >= funmap_size)
150  if (!funmap)
151  funmap = (FUNMAP **)xmalloc ((funmap_size = 80) * sizeof (FUNMAP *));
152  else
153  funmap =
154  (FUNMAP **)xrealloc (funmap, (funmap_size += 80) * sizeof (FUNMAP *));
155 
156  funmap[funmap_entry] = (FUNMAP *)xmalloc (sizeof (FUNMAP));
157  funmap[funmap_entry]->name = name;
158  funmap[funmap_entry]->function = function;
159 
160  funmap[++funmap_entry] = (FUNMAP *)NULL;
161 }

187 /* Produce a NULL terminated array of known function names. The array
188  is sorted. The array itself is allocated, but not the strings inside.
189  You should free () the array when you done, but not the pointrs. */
190 char **
191 rl_funmap_names ()
192 {
193  char **result = (char **)NULL;
194  int result_size, result_index;
195 
196  result_size = result_index = 0;
197 
198  /* Make sure that the function map has been initialized. */
199  rl_initialize_funmap ();
200 
201  for (result_index = 0; funmap[result_index]; result_index++)
202  {
203  if (result_index + 2 > result_size)
204  {
205  if (!result)
206  result = (char **)xmalloc ((result_size = 20) * sizeof (char *));
207  else
208  result = (char **)
209  xrealloc (result, (result_size += 20) * sizeof (char *));
210  }
211 
212  result[result_index] = funmap[result_index]->name;
213  result[result_index + 1] = (char *)NULL;
214  }
215 
216  qsort (result, result_index, sizeof (char *), qsort_string_compare);
217  return (result);
